// Copy and extract sources from <build server>\BD\PRKP-Integration\PRKP-Integration_<build_nr>\Sources\sources.zip
// Run tests from sources
// Remove directory names from sources (anders vindt de report generator ze niet) in de coverage output.
powershell -command "(gc <share>\CoverageReport\<build_nr>\output.xml) | % { $_ -replace 'e:[a-zA-Z\\\-\.0-9]+src\\', '' } | sc <share>\CoverageReport\<build_nr>\output.xml"
// Draai het rapport van de locatie waar ook de sources staan.
<share>\Tools\ReportGenerator\bin\ReportGenerator.exe "-reports:<share>\CoverageReport\<build_nr>\output.xml" "-targetdir:<share>\CoverageReport\<build_nr>\" " -filters:+Prkp.*;"
